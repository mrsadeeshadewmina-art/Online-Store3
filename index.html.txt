<!-- ========================= script.js ========================= -->
<script>
const products = [
  {id:1, name:"Headphones", price:50, img:"https://via.placeholder.com/200"},
  {id:2, name:"Shoes", price:80, img:"https://via.placeholder.com/200"},
  {id:3, name:"Watch", price:120, img:"https://via.placeholder.com/200"},
  {id:4, name:"Bag", price:60, img:"https://via.placeholder.com/200"}
];

let cart = JSON.parse(localStorage.getItem('cart')) || [];

const grid = document.getElementById('product-grid');

function renderProducts() {
  grid.innerHTML = '';
  products.forEach(p => {
    grid.innerHTML += `
      <div class="product" onclick="openModal(${p.id})">
        <img src="${p.img}">
        <h3>${p.name}</h3>
        <p>$${p.price}</p>
        <button onclick="addToCart(${p.id}); event.stopPropagation();">Add</button>
      </div>`;
  });
}

function addToCart(id) {
  const item = cart.find(i => i.id === id);
  if(item) item.qty++;
  else cart.push({...products.find(p=>p.id===id), qty:1});
  updateCart();
}

function updateCart() {
  localStorage.setItem('cart', JSON.stringify(cart));
  document.getElementById('cart-count').innerText = cart.reduce((a,b)=>a+b.qty,0);
  const cartItems = document.getElementById('cart-items');
  cartItems.innerHTML='';
  let total=0;
  cart.forEach(i=>{
    total += i.price*i.qty;
    cartItems.innerHTML += `<p>${i.name} x${i.qty}</p>`;
  });
  document.getElementById('cart-total').innerText = total;
}

function toggleCart(){
  document.getElementById('cart').classList.toggle('active');
}

document.getElementById('cart-btn').onclick = toggleCart;

function openModal(id){
  const p = products.find(x=>x.id===id);
  document.getElementById('modal-img').src = p.img;
  document.getElementById('modal-name').innerText = p.name;
  document.getElementById('modal-price').innerText = '$'+p.price;
  document.getElementById('modal').style.display='block';
  document.getElementById('modal-add').onclick = ()=>addToCart(id);
}

document.getElementById('close-modal').onclick = ()=>{
  document.getElementById('modal').style.display='none';
};

// Contact form validation
document.getElementById('contact-form').addEventListener('submit', function(e){
  e.preventDefault();
  alert('Message sent!');
});

renderProducts();
updateCart();
</script>
